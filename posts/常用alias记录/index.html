常用的 git aslias 记录

alias g="git"
alias gst="git status"
alias gcmsg="git commit -m"
alias gco="git checkout"
alias gcb="git checkout -b"
alias gb="git branch"
alias gp="git push"
alias gd="git diff"
alias gaa="git add ."
alias gm="git merge"
alias gcm="git checkout master"
# 合并当前分支到master
alias gmc="git branch --show-current | xargs -I {} sh -c 'git checkout master; git merge {}'"
